<script>

  const available = document.getElementById('available');

  const selected = document.getElementById('selected');

  const result = document.getElementById('result');

  // Move trait div from one container to another

  function moveTrait(trait, fromContainer, toContainer) {

    toContainer.appendChild(trait);

  }

  // Double-click handler for available traits: move to selected

  available.addEventListener('dblclick', e => {

    if (e.target && e.target.parentNode === available && e.target.matches('div[data-entrepreneur]')) {

      moveTrait(e.target, available, selected);

    }

  });

  // Double-click handler for selected traits: move back to available

  selected.addEventListener('dblclick', e => {

    if (e.target && e.target.parentNode === selected && e.target.matches('div[data-entrepreneur]')) {

      moveTrait(e.target, selected, available);

    }

  });

  // Keyboard accessibility: Enter or Space triggers double-click move

  [available, selected].forEach(container => {

    container.addEventListener('keydown', e => {

      if ((e.key === 'Enter' || e.key === ' ') && e.target && e.target.matches('div[data-entrepreneur]')) {

        e.preventDefault();

        // Simulate double-click by toggling between containers

        if (e.target.parentNode === available) {

          moveTrait(e.target, available, selected);

        } else {

          moveTrait(e.target, selected, available);

        }

      }

    });

  });

  // Check answers button

  document.getElementById('checkBtn').addEventListener('click', () => {

    const droppedTraits = Array.from(selected.querySelectorAll('div[data-entrepreneur]'));

    if (droppedTraits.length === 0) {

      setResult('Please select some traits in the Entrepreneur Traits box.', 'red');

      return;

    }

    let correctCount = 0;

    let incorrectCount = 0;

    droppedTraits.forEach(trait => {

      if (trait.dataset.entrepreneur === "true") {

        correctCount++;

      } else {

        incorrectCount++;

      }

    });

    if (incorrectCount === 0 && correctCount > 0) {

      setResult(`Great job! You identified all entrepreneur traits correctly (${correctCount} traits).`, 'green');

    } else {

      setResult(`You identified ${correctCount} entrepreneur traits correctly but included ${incorrectCount} incorrect trait(s). Try again!`, 'orange');

    }

  });

  function setResult(msg, color) {

    result.textContent = msg;

    result.style.color = color;

  }

</script>
